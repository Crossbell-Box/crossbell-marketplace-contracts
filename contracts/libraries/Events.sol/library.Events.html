AskCreated

Emitted when an ask order is created.

event AskCreated(
    address indexed owner,
    address indexed nftAddress,
    uint256 indexed tokenId,
    address payToken,
    uint256 price,
    uint256 deadline
);

Emitted when an ask order is updated.

event AskUpdated(

    uint256 indexed tokenId,
